文本
.append("text")
 .attr("x", (d, i) => d[0] + 5)
 .attr("y", (d, i) => h - d[1])
 .text((a) => a[0] + ", " + a[1])

圈：散点图
.append("circle")
.attr('cx',(d)=>d[0])
.attr('cy',(d)=> h-d[1])
.attr('r',5);

属性
.attrs({ x: 0, y: 0, width: 25, height: 100, fill: 'red' })

指数
.attr("x", (d, i) => {return i*30})

```js
<body>
  <script>
    const dataset = [12, 31, 22, 17, 25, 18, 29, 14, 9];

    const w = 500;
    const h = 100;

    const svg = d3.select("body")
                  .append("svg")
                  .attr("width", w)
                  .attr("height", h);

    svg.selectAll("rect")
       .data(dataset)
       .enter()
       .append("rect")
       .attr("x", (d, i) => i * 30)
       .attr("y", (d, i) => h - 3 * d)
       .attr("width", 25)
       .attr("height", (d, i) => 3 * d)
       .attr("fill", "navy");

    svg.selectAll("text")
       .data(dataset)
       .enter()
       .append('text')
       .attr("x", (d, i) => i * 30)
       .attr("y", (d, i) => h - 3 * d + 3)
       .text((d) => d)

  </script>
<body>
```
